		System.out.println("Derived");
	}
}

class DeriDerived extends Derived {
	public DeriDerived() {
		System.out.println("DeriDerived");
	}
}

public class Testing {
	public static void main(String[] args) {
		Derived b = new DeriDerived();
	}
}
